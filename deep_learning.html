        return min_lr
    return min_lr + 0.5 * (base_lr - min_lr) * (

class Dense:
    def __init__(self, in_dim, out_dim, rng):
        self.W = (rng.randn(in_dim, out_dim) * np.sqrt(2.0 / in_dim)).astype(np.float32)
        self.b = np.zeros((1, out_dim), dtype=np.float32)

        self.mW = np.zeros_like(self.W)
        self.vW = np.zeros_like(self.W)
        self.mb = np.zeros_like(self.b)
        self.vb = np.zeros_like(self.b)

    def forward(self, x):
        self.x = x
        return x @ self.W + self.b

    def backward(self, grad_out):
        self.dW = self.x.T @ grad_out
        self.db = np.sum(grad_out, axis=0, keepdims=True)
        grad_in = grad_out @ self.W.T
        return grad_in

    def update(self, lr, step, beta1=0.9, beta2=0.999, eps=1e-8, weight_decay=0.0):
        # AdamW風
        if weight_decay > 0.0:
            self.W *= (1.0 - lr * weight_decay)

        self.mW = beta1 * self.mW + (1.0 - beta1) * self.dW
        self.vW = beta2 * self.vW + (1.0 - beta2) * (self.dW ** 2)

        self.mb = beta1 * self.mb + (1.0 - beta1) * self.db
        self.vb = beta2 * self.vb + (1.0 - beta2) * (self.db ** 2)

        mW_hat = self.mW / (1.0 - beta1 ** step)
        vW_hat = self.vW / (1.0 - beta2 ** step)
        mb_hat = self.mb / (1.0 - beta1 ** step)
        vb_hat = self.vb / (1.0 - beta2 ** step)

        self.W -= lr * mW_hat / (np.sqrt(vW_hat) + eps)
        self.b -= lr * mb_hat / (np.sqrt(vb_hat) + eps)

    def get_state(self):
        return {
            'W': self.W.copy(),
            'b': self.b.copy()
        }

    def set_state(self, state):
        self.W = state['W'].copy()
        self.b = state['b'].copy()


class BatchNorm1d:
    def __init__(self, dim, momentum=0.9, eps=1e-5):
        self.gamma = np.ones((1, dim), dtype=np.float32)
        self.beta = np.zeros((1, dim), dtype=np.float32)

        self.running_mean = np.zeros((1, dim), dtype=np.float32)
        self.running_var = np.ones((1, dim), dtype=np.float32)

        self.momentum = momentum
        self.eps = eps

        self.mg = np.zeros_like(self.gamma)
        self.vg = np.zeros_like(self.gamma)
        self.mb = np.zeros_like(self.beta)
        self.vb = np.zeros_like(self.beta)

    def forward(self, x, training=False):
        if training:
            self.mu = np.mean(x, axis=0, keepdims=True)
            self.var = np.var(x, axis=0, keepdims=True)

            self.x_centered = x - self.mu
            self.inv_std = 1.0 / np.sqrt(self.var + self.eps)
            self.x_hat = self.x_centered * self.inv_std

            self.running_mean = self.momentum * self.running_mean + (1.0 - self.momentum) * self.mu
            self.running_var = self.momentum * self.running_var + (1.0 - self.momentum) * self.var

            return self.gamma * self.x_hat + self.beta

        x_hat = (x - self.running_mean) / np.sqrt(self.running_var + self.eps)
        return self.gamma * x_hat + self.beta

    def backward(self, grad_out):
        n = grad_out.shape[0]

        self.dgamma = np.sum(grad_out * self.x_hat, axis=0, keepdims=True)
        self.dbeta = np.sum(grad_out, axis=0, keepdims=True)

        dxhat = grad_out * self.gamma
        dvar = np.sum(
            dxhat * self.x_centered * (-0.5) * (self.inv_std ** 3),
            axis=0,
            keepdims=True
        )
        dmu = (
            np.sum(dxhat * (-self.inv_std), axis=0, keepdims=True) +
            dvar * np.mean(-2.0 * self.x_centered, axis=0, keepdims=True)
        )

        dx = (
            dxhat * self.inv_std +
            dvar * 2.0 * self.x_centered / n +
            dmu / n
        )
        return dx

    def update(self, lr, step, beta1=0.9, beta2=0.999, eps=1e-8):
        self.mg = beta1 * self.mg + (1.0 - beta1) * self.dgamma
        self.vg = beta2 * self.vg + (1.0 - beta2) * (self.dgamma ** 2)

        self.mb = beta1 * self.mb + (1.0 - beta1) * self.dbeta
        self.vb = beta2 * self.vb + (1.0 - beta2) * (self.dbeta ** 2)

        mg_hat = self.mg / (1.0 - beta1 ** step)
        vg_hat = self.vg / (1.0 - beta2 ** step)
        mb_hat = self.mb / (1.0 - beta1 ** step)
        vb_hat = self.vb / (1.0 - beta2 ** step)

        self.gamma -= lr * mg_hat / (np.sqrt(vg_hat) + eps)
        self.beta -= lr * mb_hat / (np.sqrt(vb_hat) + eps)

    def get_state(self):
        return {
            'gamma': self.gamma.copy(),
            'beta': self.beta.copy(),
            'running_mean': self.running_mean.copy(),
            'running_var': self.running_var.copy()
        }

    def set_state(self, state):
        self.gamma = state['gamma'].copy()
        self.beta = state['beta'].copy()
        self.running_mean = state['running_mean'].copy()
        self.running_var = state['running_var'].copy()


class Dropout:
    def __init__(self, p, rng):
        self.p = p
        self.rng = rng
        self.mask = None

    def forward(self, x, training=False):
        if training and self.p > 0.0:
            self.mask = (
                (self.rng.rand(*x.shape) >= self.p).astype(np.float32) / (1.0 - self.p)
            )
            return x * self.mask

        self.mask = None
        return x

    def backward(self, grad_out):
        if self.mask is None:
            return grad_out
        return grad_out * self.mask


class Model:
    def __init__(self, rng, input_dim=784, hidden_dims=(1024, 512, 256), output_dim=10,
                 dropout_rates=(0.10, 0.05, 0.02)):
        h1, h2, h3 = hidden_dims
        d1, d2, d3 = dropout_rates

        self.fc1 = Dense(input_dim, h1, rng)
        self.bn1 = BatchNorm1d(h1)
        self.do1 = Dropout(d1, rng)

        self.fc2 = Dense(h1, h2, rng)
        self.bn2 = BatchNorm1d(h2)
        self.do2 = Dropout(d2, rng)

        self.fc3 = Dense(h2, h3, rng)
        self.bn3 = BatchNorm1d(h3)
        self.do3 = Dropout(d3, rng)

        self.fc4 = Dense(h3, output_dim, rng)

        self.global_step = 0

    def __call__(self, x, training=False):
        self.z1 = self.fc1.forward(x)
        self.n1 = self.bn1.forward(self.z1, training=training)
        self.h1 = leaky_relu(self.n1)
        self.h1 = self.do1.forward(self.h1, training=training)

        self.z2 = self.fc2.forward(self.h1)
        self.n2 = self.bn2.forward(self.z2, training=training)
        self.h2 = leaky_relu(self.n2)
        self.h2 = self.do2.forward(self.h2, training=training)

        self.z3 = self.fc3.forward(self.h2)
        self.n3 = self.bn3.forward(self.z3, training=training)
        self.h3 = leaky_relu(self.n3)
        self.h3 = self.do3.forward(self.h3, training=training)

        self.z4 = self.fc4.forward(self.h3)
        self.y = softmax(self.z4)
        return self.y

    def backward(self, t):
        delta = (self.y - t) / t.shape[0]

        delta = self.fc4.backward(delta)

        delta = self.do3.backward(delta)
        delta = delta * deriv_leaky_relu(self.n3)
        delta = self.bn3.backward(delta)
        delta = self.fc3.backward(delta)

        delta = self.do2.backward(delta)
        delta = delta * deriv_leaky_relu(self.n2)
        delta = self.bn2.backward(delta)
        delta = self.fc2.backward(delta)

        delta = self.do1.backward(delta)
        delta = delta * deriv_leaky_relu(self.n1)
        delta = self.bn1.backward(delta)
        _ = self.fc1.backward(delta)

    def update(self, lr, weight_decay=0.0):
        self.global_step += 1

        self.fc1.update(lr, self.global_step, weight_decay=weight_decay)
        self.bn1.update(lr, self.global_step)

        self.fc2.update(lr, self.global_step, weight_decay=weight_decay)
        self.bn2.update(lr, self.global_step)

        self.fc3.update(lr, self.global_step, weight_decay=weight_decay)
        self.bn3.update(lr, self.global_step)

        self.fc4.update(lr, self.global_step, weight_decay=weight_decay)

    def get_state(self):
        return {
            'fc1': self.fc1.get_state(),
            'bn1': self.bn1.get_state(),
            'fc2': self.fc2.get_state(),
            'bn2': self.bn2.get_state(),
            'fc3': self.fc3.get_state(),
            'bn3': self.bn3.get_state(),
            'fc4': self.fc4.get_state(),
            'global_step': self.global_step
        }

    def set_state(self, state):
        self.fc1.set_state(state['fc1'])
        self.bn1.set_state(state['bn1'])
        self.fc2.set_state(state['fc2'])
        self.bn2.set_state(state['bn2'])
        self.fc3.set_state(state['fc3'])
        self.bn3.set_state(state['bn3'])
        self.fc4.set_state(state['fc4'])
        self.global_step = state['global_step']

def shift_batch(x_raw, dy=0, dx=0):
    n = x_raw.shape[0]
    imgs = x_raw.reshape(n, 28, 28)

    pad_y = abs(dy)
    pad_x = abs(dx)

    padded = np.pad(
        imgs,
        ((0, 0), (pad_y, pad_y), (pad_x, pad_x)),
        mode='constant',
        constant_values=0.0
    )

    sy = pad_y + dy
    sx = pad_x + dx

    shifted = padded[:, sy:sy + 28, sx:sx + 28]
    return shifted.reshape(n, -1).astype(np.float32)


def random_shift_batch(x_raw, rng, max_shift=2):
    n = x_raw.shape[0]
    imgs = x_raw.reshape(n, 28, 28)

    padded = np.pad(
        imgs,
        ((0, 0), (max_shift, max_shift), (max_shift, max_shift)),
        mode='constant',
        constant_values=0.0
    )

    out = np.empty_like(imgs)
    shifts_y = rng.randint(-max_shift, max_shift + 1, size=n)
    shifts_x = rng.randint(-max_shift, max_shift + 1, size=n)

    for i, (dy, dx) in enumerate(zip(shifts_y, shifts_x)):
        sy = dy + max_shift
        sx = dx + max_shift
        out[i] = padded[i, sy:sy + 28, sx:sx + 28]

    return out.reshape(n, -1).astype(np.float32)


def predict_proba(model, x_std, batch_size=512):
    probs = []
    for start in range(0, x_std.shape[0], batch_size):
        xb = x_std[start:start + batch_size]
        probs.append(model(xb, training=False))
    return np.vstack(probs).astype(np.float32)


def predict_proba_tta(model, x_raw, batch_size=512, shifts=None):
    if shifts is None:
        shifts = [(0, 0)]

    avg_prob = np.zeros((x_raw.shape[0], 10), dtype=np.float32)

    for dy, dx in shifts:
        if dy == 0 and dx == 0:
            shifted = x_raw
        else:
            shifted = shift_batch(x_raw, dy=dy, dx=dx)

        shifted_std = standardize(shifted)
        avg_prob += predict_proba(model, shifted_std, batch_size=batch_size)

    avg_prob /= len(shifts)
    return avg_prob.astype(np.float32)


def train_single_model(
    config,
    x_train_raw,
    t_train,
    x_val_std,
    t_val,
    n_epochs=30,
    batch_size=192,
    weight_decay=3e-5,
    label_smoothing=0.02,
    patience=5,
    max_shift=2,
    min_lr=1e-4
):
    rng = np.random.RandomState(config['seed'])

    model = Model(
        rng=rng,
        input_dim=784,
        hidden_dims=config['hidden_dims'],
        output_dim=10,
        dropout_rates=config['dropout_rates']
    )

    best_state = model.get_state()
    best_val_acc = -1.0
    best_epoch = 0
    wait = 0

    for epoch in range(n_epochs):
        current_lr = cosine_lr(
            epoch=epoch,
            total_epochs=n_epochs,
            base_lr=config['base_lr'],
            min_lr=min_lr
        )

        perm = rng.permutation(x_train_raw.shape[0])

        train_loss_sum = 0.0
        train_correct = 0
        train_count = 0

        for start in range(0, x_train_raw.shape[0], batch_size):
            idx = perm[start:start + batch_size]

            xb_raw = x_train_raw[idx]
            tb = t_train[idx]

            xb_raw = random_shift_batch(xb_raw, rng, max_shift=max_shift)
            xb = standardize(xb_raw)

            tb_smooth = smooth_labels(tb, eps=label_smoothing)

            y = model(xb, training=True)
            loss = crossentropy_loss(tb_smooth, y)

            model.backward(tb_smooth)
            model.update(current_lr, weight_decay=weight_decay)

            train_loss_sum += loss * xb.shape[0]
            train_correct += np.sum(np.argmax(tb, axis=1) == np.argmax(y, axis=1))
            train_count += xb.shape[0]

        train_loss = train_loss_sum / train_count
        train_acc = train_correct / train_count

        y_val = predict_proba(model, x_val_std, batch_size=512)
        val_loss = crossentropy_loss(t_val, y_val)
        val_acc = accuracy_numpy(t_val, y_val)

        print(
            'seed={} epoch={:02d} lr={:.6f} train_loss={:.4f} train_acc={:.4f} val_loss={:.4f} val_acc={:.4f}'.format(
                config['seed'],
                epoch + 1,
                current_lr,
                train_loss,
                train_acc,
                val_loss,
                val_acc
            )
        )

        if val_acc > best_val_acc + 1e-5:
            best_val_acc = val_acc
            best_epoch = epoch + 1
            best_state = model.get_state()
            wait = 0
        else:
            wait += 1
            if wait >= patience:
                print(
                    'early stopping: seed={} best_epoch={} best_val_acc={:.4f}'.format(
                        config['seed'],
                        best_epoch,
                        best_val_acc
                    )
                )
                break

    model.set_state(best_state)
    return model, best_epoch, best_val_acc

for cfg in ensemble_configs:
    model_name = 'seed{}_h{}_{}_{}_ep{}_bs{}'.format(

        best_val_acc = float(data['best_val_acc'])
        best_epoch = int(data['best_epoch'])

        print(
            'Loaded cache: {} | best_epoch={} | best_val_acc={:.4f}'.format(

    else:
        model, best_epoch, best_val_acc = train_single_model(

        )

        # best modelでだけTTA推論
        val_prob = predict_proba_tta(

        )

        test_prob = predict_proba_tta(
            model,
