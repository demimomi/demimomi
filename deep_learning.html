import numpy as np
import pandas as pd
from sklearn.model_selection import train_test_split

# =========================
# load
# =========================

# =========================
# feature engineering
# 単層モデルのまま、前処理だけ強くする
# =========================
def _make_features_batch(x_batch, cell_size=4, bins=9):
    img = x_batch.reshape(-1, 28, 28).astype(np.float32)
    n = img.shape[0]

    # 1) 2x2 average pooling -> 14x14 = 196 dims
    pooled = img.reshape(n, 14, 2, 14, 2).mean(axis=(2, 4)).reshape(n, -1)

    # 2) row / col projection = 28 + 28 dims
    row_proj = img.mean(axis=2)
    col_proj = img.mean(axis=1)

    # 3) HOG-like feature
    gx = np.empty_like(img)
    gy = np.empty_like(img)

    gx[:, :, 1:-1] = img[:, :, 2:] - img[:, :, :-2]
    gx[:, :, 0] = img[:, :, 1] - img[:, :, 0]
    gx[:, :, -1] = img[:, :, -1] - img[:, :, -2]

    gy[:, 1:-1, :] = img[:, 2:, :] - img[:, :-2, :]
    gy[:, 0, :] = img[:, 1, :] - img[:, 0, :]
    gy[:, -1, :] = img[:, -1, :] - img[:, -2, :]

    mag = np.sqrt(gx * gx + gy * gy + 1e-12).astype(np.float32)

    ori = (np.degrees(np.arctan2(gy, gx)) + 180.0) % 180.0
    ori = ori.astype(np.float32)

    bin_width = 180.0 / bins
    bin_pos = ori / bin_width

    b0 = np.floor(bin_pos).astype(np.int32) % bins
    b1 = (b0 + 1) % bins
    w1 = bin_pos - np.floor(bin_pos)
    w0 = 1.0 - w1

    ncy = 28 // cell_size
    ncx = 28 // cell_size
    hog = np.zeros((n, ncy, ncx, bins), dtype=np.float32)

    for cy in range(ncy):
        ys = cy * cell_size
        ye = ys + cell_size
        for cx in range(ncx):
            xs = cx * cell_size
            xe = xs + cell_size

            cell_mag = mag[:, ys:ye, xs:xe]
            cell_b0 = b0[:, ys:ye, xs:xe]
            cell_b1 = b1[:, ys:ye, xs:xe]
            cell_w0 = w0[:, ys:ye, xs:xe]
            cell_w1 = w1[:, ys:ye, xs:xe]

            hist = np.zeros((n, bins), dtype=np.float32)

            for b in range(bins):
                hist[:, b] = np.sum(
                    np.where(cell_b0 == b, cell_mag * cell_w0, 0.0) +
                    np.where(cell_b1 == b, cell_mag * cell_w1, 0.0),
                    axis=(1, 2)
                )

            hog[:, cy, cx, :] = hist

    hog = hog.reshape(n, -1)

    # L2 normalize HOG per sample
    hog_norm = np.sqrt(np.sum(hog * hog, axis=1, keepdims=True) + 1e-6)
    hog = hog / hog_norm

    # final feature
    feat = np.concatenate([pooled, hog, row_proj, col_proj], axis=1).astype(np.float32)
    return feat

def make_features(x, batch_size=2000):
    feats = []
    for i in range(0, len(x), batch_size):
        feats.append(_make_features_batch(x[i:i + batch_size]))
    return np.vstack(feats).astype(np.float32)

def standardize_fit(X):
    mean = X.mean(axis=0, keepdims=True)
    std = X.std(axis=0, keepdims=True)
    std[std < 1e-6] = 1.0
    return mean.astype(np.float32), std.astype(np.float32)

def standardize_apply(X, mean, std):
    X = (X - mean) / std
    X = np.clip(X, -5.0, 5.0)
    return X.astype(np.float32)

# =========================
# softmax regression (numpy only)
# =========================
def softmax(z):
    z = z - np.max(z, axis=1, keepdims=True)
    np.exp(z, out=z)
    z /= np.sum(z, axis=1, keepdims=True)
    return z

def cross_entropy(y_true, y_prob):
    eps = 1e-12
    return -np.mean(np.sum(y_true * np.log(y_prob + eps), axis=1))

def accuracy(y_true, y_prob):
    return np.mean(np.argmax(y_true, axis=1) == np.argmax(y_prob, axis=1))

def train_softmax(
    X_train, y_train,
    X_valid=None, y_valid=None,
    epochs=100,
    batch_size=256,
    lr_max=2e-3,
    lr_min=2e-4,
    weight_decay=1e-4,
    label_smoothing=0.02,
    patience=15,
    seed=42
):
    rng = np.random.RandomState(seed)
    n, d = X_train.shape
    k = y_train.shape[1]

    W = (0.01 * rng.randn(d, k)).astype(np.float32)
    b = np.zeros((1, k), dtype=np.float32)

    mW = np.zeros_like(W)
    vW = np.zeros_like(W)
    mb = np.zeros_like(b)
    vb = np.zeros_like(b)

    beta1 = 0.9
    beta2 = 0.999
    eps = 1e-8

    steps_per_epoch = (n + batch_size - 1) // batch_size
    total_steps = max(1, epochs * steps_per_epoch)
    global_step = 0

    best_val_acc = -1.0
    best_epoch = 1
    best_W = W.copy()
    best_b = b.copy()
    wait = 0

    for epoch in range(1, epochs + 1):
        perm = rng.permutation(n)
        Xs = X_train[perm]
        ys = y_train[perm]

        for start in range(0, n, batch_size):
            xb = Xs[start:start + batch_size]
            yb = ys[start:start + batch_size]
            bs = xb.shape[0]

            if label_smoothing > 0.0:
                yb_used = yb * (1.0 - label_smoothing) + (label_smoothing / k)
            else:
                yb_used = yb

            logits = xb @ W + b
            prob = softmax(logits)

            dlogits = (prob - yb_used) / bs
            dW = xb.T @ dlogits
            db = np.sum(dlogits, axis=0, keepdims=True)

            global_step += 1
            t = global_step

            # cosine decay
            lr = lr_min + 0.5 * (lr_max - lr_min) * (
                1.0 + np.cos(np.pi * (t - 1) / total_steps)
            )

            # Adam
            mW = beta1 * mW + (1.0 - beta1) * dW
            vW = beta2 * vW + (1.0 - beta2) * (dW * dW)
            mb = beta1 * mb + (1.0 - beta1) * db
            vb = beta2 * vb + (1.0 - beta2) * (db * db)

            mW_hat = mW / (1.0 - beta1 ** t)
            vW_hat = vW / (1.0 - beta2 ** t)
            mb_hat = mb / (1.0 - beta1 ** t)
            vb_hat = vb / (1.0 - beta2 ** t)

            # decoupled weight decay
            W *= (1.0 - lr * weight_decay)

            W -= lr * mW_hat / (np.sqrt(vW_hat) + eps)
            b -= lr * mb_hat / (np.sqrt(vb_hat) + eps)

        train_prob = softmax(X_train @ W + b)
        train_loss = cross_entropy(y_train, train_prob)
        train_acc = accuracy(y_train, train_prob)

        if X_valid is not None:
            valid_prob = softmax(X_valid @ W + b)
            valid_loss = cross_entropy(y_valid, valid_prob)
            valid_acc = accuracy(y_valid, valid_prob)

            if valid_acc > best_val_acc:
                best_val_acc = valid_acc
                best_epoch = epoch
                best_W = W.copy()
                best_b = b.copy()
                wait = 0
            else:
                wait += 1

            if epoch == 1 or epoch % 5 == 0:
                print(
                    f"Epoch {epoch:3d} | "
                    f"train_loss={train_loss:.4f} train_acc={train_acc:.4f} | "
                    f"valid_loss={valid_loss:.4f} valid_acc={valid_acc:.4f}"
                )

            if wait >= patience:
                print(f"Early stopping at epoch {epoch}")
                break
        else:
            if epoch == 1 or epoch % 5 == 0 or epoch == epochs:
                print(
                    f"Epoch {epoch:3d} | "
                    f"train_loss={train_loss:.4f} train_acc={train_acc:.4f}"
                )

    if X_valid is not None:
        return {
            "W": best_W,
            "b": best_b,
            "best_epoch": best_epoch,
            "best_val_acc": best_val_acc,
        }
    else:
        return {
            "W": W,
            "b": b,
            "best_epoch": epochs,
            "best_val_acc": None,
        }

# =========================
# 1) feature extraction
# =========================
X_all = make_features(x_train, batch_size=2000)
X_test_feat = make_features(x_test, batch_size=2000)

# =========================
# 2) validation split
# =========================
y_labels = np.argmax(y_train, axis=1)
idx = np.arange(len(X_all))

idx_tr, idx_va = train_test_split(
    idx,
    test_size=0.1,
    random_state=42,
    stratify=y_labels
)

X_tr_raw = X_all[idx_tr]
X_va_raw = X_all[idx_va]
y_tr = y_train[idx_tr]
y_va = y_train[idx_va]

mean, std = standardize_fit(X_tr_raw)
X_tr = standardize_apply(X_tr_raw, mean, std)
X_va = standardize_apply(X_va_raw, mean, std)

# =========================
# 3) stage 1: best epoch search
# =========================
stage1 = train_softmax(
    X_tr, y_tr,
    X_valid=X_va, y_valid=y_va,
    epochs=100,
    batch_size=256,
    lr_max=2e-3,
    lr_min=2e-4,
    weight_decay=1e-4,
    label_smoothing=0.02,
    patience=15,
    seed=42
)

print("best_epoch =", stage1["best_epoch"])
print("best_val_acc =", stage1["best_val_acc"])

# =========================
# 4) stage 2: retrain on full training set
# =========================
mean_full, std_full = standardize_fit(X_all)
X_full = standardize_apply(X_all, mean_full, std_full)
X_test_std = standardize_apply(X_test_feat, mean_full, std_full)

final_model = train_softmax(
    X_full, y_train,
    X_valid=None, y_valid=None,
    epochs=stage1["best_epoch"],
    batch_size=256,
    lr_max=2e-3,
    lr_min=2e-4,
    weight_decay=1e-4,
    label_smoothing=0.02,
    patience=15,
    seed=42
)

# =========================
# 5) prediction / submission
# =========================

y_pred = np.argmax(test_prob, axis=1)

submission = pd.Series(y_pred, name="label")

    work_dir + "/Lecture02/submission_pred_02.csv",

これは Lecture3 の homework として作成された Python コードで、このコードは、Fashion MNIST という衣類画像データを、多層パーセプトロン（MLP）によって 10 クラスに分類する課題に対応したものです。扱う画像は 28×28 ピクセルの白黒画像で、各画像が「Tシャツ」「靴」「バッグ」などのどれに当たるかを判定します。

まず前半では、Google Drive 上の学習データとテストデータを読み込み、画像を 0〜1 に正規化したうえで 784 次元のベクトルに変換しています。正解ラベルは one-hot 形式に変換し、さらに学習データを訓練用と検証用に分割して、訓練データの平均と標準偏差を使って標準化しています。これにより、各入力次元のスケールが揃い、学習が安定しやすくなっています。

モデル本体は NumPy のみで実装された 4 層構成の MLP で、入力 784 次元から 1024 → 512 → 256 → 10 と変換していきます。各中間層では ReLU を使い、さらに Dropout を入れることで過学習を抑えています。重みの初期値には He 初期化を使い、出力層では softmax によって各クラスの確率を計算します。逆伝播では、softmax とクロスエントロピー誤差の簡約形を使って勾配を効率よく計算しています。

学習では、ミニバッチごとに損失を計算し、AdamW 風の更新則で重みを調整しています。学習率はエポックが進むにつれて少しずつ下げ、weight decay も加えることで汎化性能を高めています。また、検証精度が改善しなくなった時点で Early Stopping を行い、最も良かった時点のパラメータを保存するようにしています。

最後に、その最良モデルを使ってテストデータをバッチ単位で推論し、予測ラベルを CSV 形式で保存しています。要するにこのコードは、「多層パーセプトロンを NumPy のみで実装する」という条件の中で、標準化・Dropout・最適化手法・学習率制御・Early Stopping を組み合わせ、精度と安定性の両方を高めた実装です。

def np_log(x):
    return np.log(np.clip(x, 1e-10, 1e10))


def create_batch(data, batch_size):
    batches = []
    for start in range(0, data.shape[0], batch_size):
        end = start + batch_size
        batches.append(data[start:end])
    return batches


def relu(x):
    return np.maximum(0.0, x)


def deriv_relu(x):
    return (x > 0).astype(np.float32)


def softmax(x):
    x = x - np.max(x, axis=1, keepdims=True)  # 数値安定化
    exp_x = np.exp(x)
    return exp_x / np.sum(exp_x, axis=1, keepdims=True)


def deriv_softmax(y):
    # 今回の学習では softmax + cross entropy の簡約形を使うため未使用
    # ただし関数枠があるので、正しい Jacobian を返す形で実装
    if y.ndim == 1:
        y = y[np.newaxis, :]

    jacobians = []
    for i in range(y.shape[0]):
        s = y[i].reshape(-1, 1)
        jacobians.append(np.diagflat(s) - s @ s.T)
    return np.array(jacobians)


def crossentropy_loss(t, y):
    return -np.sum(t * np_log(y)) / t.shape[0]


def accuracy_numpy(t, y):
    return np.mean(np.argmax(t, axis=1) == np.argmax(y, axis=1))


class Dense:
    def __init__(self, in_dim, out_dim, rng):
        # He initialization
        self.W = (rng.randn(in_dim, out_dim) * np.sqrt(2.0 / in_dim)).astype(np.float32)
        self.b = np.zeros((1, out_dim), dtype=np.float32)

        # Adam states
        self.mW = np.zeros_like(self.W)
        self.vW = np.zeros_like(self.W)
        self.mb = np.zeros_like(self.b)
        self.vb = np.zeros_like(self.b)

    def forward(self, x):
        self.x = x
        return x @ self.W + self.b

    def backward(self, grad_out):
        self.dW = self.x.T @ grad_out
        self.db = np.sum(grad_out, axis=0, keepdims=True)
        grad_in = grad_out @ self.W.T
        return grad_in

    def update(self, lr, t, beta1=0.9, beta2=0.999, eps=1e-8, weight_decay=0.0):
        # AdamW 風の decoupled weight decay
        if weight_decay > 0.0:
            self.W *= (1.0 - lr * weight_decay)

        self.mW = beta1 * self.mW + (1.0 - beta1) * self.dW
        self.vW = beta2 * self.vW + (1.0 - beta2) * (self.dW ** 2)

        self.mb = beta1 * self.mb + (1.0 - beta1) * self.db
        self.vb = beta2 * self.vb + (1.0 - beta2) * (self.db ** 2)

        mW_hat = self.mW / (1.0 - beta1 ** t)
        vW_hat = self.vW / (1.0 - beta2 ** t)

        mb_hat = self.mb / (1.0 - beta1 ** t)
        vb_hat = self.vb / (1.0 - beta2 ** t)

        self.W -= lr * mW_hat / (np.sqrt(vW_hat) + eps)
        self.b -= lr * mb_hat / (np.sqrt(vb_hat) + eps)


class Dropout:
    def __init__(self, p, rng):
        self.p = p
        self.rng = rng
        self.mask = None

    def forward(self, x, training=False):
        if training and self.p > 0.0:
            self.mask = (self.rng.rand(*x.shape) >= self.p).astype(np.float32) / (1.0 - self.p)
            return x * self.mask
        self.mask = None
        return x

    def backward(self, grad_out):
        if self.mask is None:
            return grad_out
        return grad_out * self.mask


class Model:
    def __init__(self, rng, input_dim=784, hidden_dims=(1024, 512, 256), output_dim=10,
                 dropout_rates=(0.25, 0.20, 0.10)):
        h1, h2, h3 = hidden_dims
        d1, d2, d3 = dropout_rates

        self.fc1 = Dense(input_dim, h1, rng)
        self.fc2 = Dense(h1, h2, rng)
        self.fc3 = Dense(h2, h3, rng)
        self.fc4 = Dense(h3, output_dim, rng)

        self.do1 = Dropout(d1, rng)
        self.do2 = Dropout(d2, rng)
        self.do3 = Dropout(d3, rng)

        self.global_step = 0

    def __call__(self, x, training=False):
        self.z1 = self.fc1.forward(x)
        self.h1 = relu(self.z1)
        self.h1 = self.do1.forward(self.h1, training=training)

        self.z2 = self.fc2.forward(self.h1)
        self.h2 = relu(self.z2)
        self.h2 = self.do2.forward(self.h2, training=training)

        self.z3 = self.fc3.forward(self.h2)
        self.h3 = relu(self.z3)
        self.h3 = self.do3.forward(self.h3, training=training)

        self.z4 = self.fc4.forward(self.h3)
        self.y = softmax(self.z4)
        return self.y

    def backward(self, t):
        batch_size = t.shape[0]

        # softmax + cross entropy の簡約微分
        delta = (self.y - t) / batch_size

        delta = self.fc4.backward(delta)

        delta = self.do3.backward(delta)
        delta = delta * deriv_relu(self.z3)
        delta = self.fc3.backward(delta)

        delta = self.do2.backward(delta)
        delta = delta * deriv_relu(self.z2)
        delta = self.fc2.backward(delta)

        delta = self.do1.backward(delta)
        delta = delta * deriv_relu(self.z1)
        _ = self.fc1.backward(delta)

    def update(self, lr, weight_decay=0.0):
        self.global_step += 1
        for layer in [self.fc1, self.fc2, self.fc3, self.fc4]:
            layer.update(
                lr=lr,
                t=self.global_step,
                weight_decay=weight_decay
            )

    def get_parameters(self):
        return [
            (self.fc1.W.copy(), self.fc1.b.copy()),
            (self.fc2.W.copy(), self.fc2.b.copy()),
            (self.fc3.W.copy(), self.fc3.b.copy()),
            (self.fc4.W.copy(), self.fc4.b.copy()),
        ]

    def set_parameters(self, params):
        (self.fc1.W, self.fc1.b) = params[0]
        (self.fc2.W, self.fc2.b) = params[1]
        (self.fc3.W, self.fc3.b) = params[2]
        (self.fc4.W, self.fc4.b) = params[3]

def train_model(
    mlp,
    x_train,
    t_train,
    x_val,
    t_val,
    n_epochs=40,
    batch_size=128,
    lr=1e-3,
    weight_decay=1e-4,
    patience=8
):
    best_val_acc = -1.0
    best_epoch = 0
    best_params = mlp.get_parameters()
    wait = 0

    for epoch in range(n_epochs):
        losses_train = []
        losses_valid = []
        train_num = 0
        train_true_num = 0
        valid_num = 0
        valid_true_num = 0

        # 学習率を少しずつ下げる
        current_lr = max(lr * (0.97 ** epoch), 2e-4)

        # Shuffle training data
        perm = rng.permutation(x_train.shape[0])
        x_train_epoch = x_train[perm]
        t_train_epoch = t_train[perm]

        x_train_batches = create_batch(x_train_epoch, batch_size)
        t_train_batches = create_batch(t_train_epoch, batch_size)

        x_val_batches = create_batch(x_val, batch_size)
        t_val_batches = create_batch(t_val, batch_size)

        # Train
        for x, t in zip(x_train_batches, t_train_batches):
            y = mlp(x, training=True)

            loss = crossentropy_loss(t, y)
            losses_train.append(loss)

            mlp.backward(t)
            mlp.update(current_lr, weight_decay=weight_decay)

            train_true_num += np.sum(np.argmax(t, axis=1) == np.argmax(y, axis=1))
            train_num += x.shape[0]

        # Validation
        for x, t in zip(x_val_batches, t_val_batches):
            y = mlp(x, training=False)

            loss = crossentropy_loss(t, y)
            losses_valid.append(loss)

            valid_true_num += np.sum(np.argmax(t, axis=1) == np.argmax(y, axis=1))
            valid_num += x.shape[0]

        train_loss = np.mean(losses_train)
        train_acc = train_true_num / train_num
        valid_loss = np.mean(losses_valid)
        valid_acc = valid_true_num / valid_num

        print(
            "EPOCH: {}, LR: {:.6f}, Train [Loss: {:.4f}, Accuracy: {:.4f}], "
            "Valid [Loss: {:.4f}, Accuracy: {:.4f}]".format(
                epoch + 1,
                current_lr,
                train_loss,
                train_acc,
                valid_loss,
                valid_acc
            )
        )

        # best model を保存
        if valid_acc > best_val_acc:
            best_val_acc = valid_acc
            best_epoch = epoch + 1
            best_params = mlp.get_parameters()
            wait = 0
        else:
            wait += 1
            if wait >= patience:
                print("Early stopping at epoch {}.".format(epoch + 1))
                break

    mlp.set_parameters(best_params)
    print("Best epoch: {}, Best valid accuracy: {:.4f}".format(best_epoch, best_val_acc))


train_model(
    mlp,
    x_train,
    t_train,
    x_val,
    t_val,
    n_epochs=n_epochs,
    batch_size=batch_size,
    lr=lr,
    weight_decay=weight_decay,
    patience=patience
)

# Test prediction
