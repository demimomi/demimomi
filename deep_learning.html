"""### ソフトマックス回帰の実装

### Implementing softmax regression

import numpy as np
import pandas as pd
from sklearn.model_selection import train_test_split

# =========================
# load
# =========================
x_train, y_train, x_test = load_fashionmnist()

# =========================
# feature engineering
# 単層モデルのまま、前処理だけ強くする
# =========================
def _make_features_batch(x_batch, cell_size=4, bins=9):
    img = x_batch.reshape(-1, 28, 28).astype(np.float32)
    n = img.shape[0]

    # 1) 2x2 average pooling -> 14x14 = 196 dims
    pooled = img.reshape(n, 14, 2, 14, 2).mean(axis=(2, 4)).reshape(n, -1)

    # 2) row / col projection = 28 + 28 dims
    row_proj = img.mean(axis=2)
    col_proj = img.mean(axis=1)

    # 3) HOG-like feature
    gx = np.empty_like(img)
    gy = np.empty_like(img)

    gx[:, :, 1:-1] = img[:, :, 2:] - img[:, :, :-2]
    gx[:, :, 0] = img[:, :, 1] - img[:, :, 0]
    gx[:, :, -1] = img[:, :, -1] - img[:, :, -2]

    gy[:, 1:-1, :] = img[:, 2:, :] - img[:, :-2, :]
    gy[:, 0, :] = img[:, 1, :] - img[:, 0, :]
    gy[:, -1, :] = img[:, -1, :] - img[:, -2, :]

    mag = np.sqrt(gx * gx + gy * gy + 1e-12).astype(np.float32)

    ori = (np.degrees(np.arctan2(gy, gx)) + 180.0) % 180.0
    ori = ori.astype(np.float32)

    bin_width = 180.0 / bins
    bin_pos = ori / bin_width

    b0 = np.floor(bin_pos).astype(np.int32) % bins
    b1 = (b0 + 1) % bins
    w1 = bin_pos - np.floor(bin_pos)
    w0 = 1.0 - w1

    ncy = 28 // cell_size
    ncx = 28 // cell_size
    hog = np.zeros((n, ncy, ncx, bins), dtype=np.float32)

    for cy in range(ncy):
        ys = cy * cell_size
        ye = ys + cell_size
        for cx in range(ncx):
            xs = cx * cell_size
            xe = xs + cell_size

            cell_mag = mag[:, ys:ye, xs:xe]
            cell_b0 = b0[:, ys:ye, xs:xe]
            cell_b1 = b1[:, ys:ye, xs:xe]
            cell_w0 = w0[:, ys:ye, xs:xe]
            cell_w1 = w1[:, ys:ye, xs:xe]

            hist = np.zeros((n, bins), dtype=np.float32)

            for b in range(bins):
                hist[:, b] = np.sum(
                    np.where(cell_b0 == b, cell_mag * cell_w0, 0.0) +
                    np.where(cell_b1 == b, cell_mag * cell_w1, 0.0),
                    axis=(1, 2)
                )

            hog[:, cy, cx, :] = hist

    hog = hog.reshape(n, -1)

    # L2 normalize HOG per sample
    hog_norm = np.sqrt(np.sum(hog * hog, axis=1, keepdims=True) + 1e-6)
    hog = hog / hog_norm

    # final feature
    feat = np.concatenate([pooled, hog, row_proj, col_proj], axis=1).astype(np.float32)
    return feat

def make_features(x, batch_size=2000):
    feats = []
    for i in range(0, len(x), batch_size):
        feats.append(_make_features_batch(x[i:i + batch_size]))
    return np.vstack(feats).astype(np.float32)

def standardize_fit(X):
    mean = X.mean(axis=0, keepdims=True)
    std = X.std(axis=0, keepdims=True)
    std[std < 1e-6] = 1.0
    return mean.astype(np.float32), std.astype(np.float32)

def standardize_apply(X, mean, std):
    X = (X - mean) / std
    X = np.clip(X, -5.0, 5.0)
    return X.astype(np.float32)

# =========================
# softmax regression (numpy only)
# =========================
def softmax(z):
    z = z - np.max(z, axis=1, keepdims=True)
    np.exp(z, out=z)
    z /= np.sum(z, axis=1, keepdims=True)
    return z

def cross_entropy(y_true, y_prob):
    eps = 1e-12
    return -np.mean(np.sum(y_true * np.log(y_prob + eps), axis=1))

def accuracy(y_true, y_prob):
    return np.mean(np.argmax(y_true, axis=1) == np.argmax(y_prob, axis=1))

def train_softmax(
    X_train, y_train,
    X_valid=None, y_valid=None,
    epochs=100,
    batch_size=256,
    lr_max=2e-3,
    lr_min=2e-4,
    weight_decay=1e-4,
    label_smoothing=0.02,
    patience=15,
    seed=42
):
    rng = np.random.RandomState(seed)
    n, d = X_train.shape
    k = y_train.shape[1]

    W = (0.01 * rng.randn(d, k)).astype(np.float32)
    b = np.zeros((1, k), dtype=np.float32)

    mW = np.zeros_like(W)
    vW = np.zeros_like(W)
    mb = np.zeros_like(b)
    vb = np.zeros_like(b)

    beta1 = 0.9
    beta2 = 0.999
    eps = 1e-8

    steps_per_epoch = (n + batch_size - 1) // batch_size
    total_steps = max(1, epochs * steps_per_epoch)
    global_step = 0

    best_val_acc = -1.0
    best_epoch = 1
    best_W = W.copy()
    best_b = b.copy()
    wait = 0

    for epoch in range(1, epochs + 1):
        perm = rng.permutation(n)
        Xs = X_train[perm]
        ys = y_train[perm]

        for start in range(0, n, batch_size):
            xb = Xs[start:start + batch_size]
            yb = ys[start:start + batch_size]
            bs = xb.shape[0]

            if label_smoothing > 0.0:
                yb_used = yb * (1.0 - label_smoothing) + (label_smoothing / k)
            else:
                yb_used = yb

            logits = xb @ W + b
            prob = softmax(logits)

            dlogits = (prob - yb_used) / bs
            dW = xb.T @ dlogits
            db = np.sum(dlogits, axis=0, keepdims=True)

            global_step += 1
            t = global_step

            # cosine decay
            lr = lr_min + 0.5 * (lr_max - lr_min) * (
                1.0 + np.cos(np.pi * (t - 1) / total_steps)
            )

            # Adam
            mW = beta1 * mW + (1.0 - beta1) * dW
            vW = beta2 * vW + (1.0 - beta2) * (dW * dW)
            mb = beta1 * mb + (1.0 - beta1) * db
            vb = beta2 * vb + (1.0 - beta2) * (db * db)

            mW_hat = mW / (1.0 - beta1 ** t)
            vW_hat = vW / (1.0 - beta2 ** t)
            mb_hat = mb / (1.0 - beta1 ** t)
            vb_hat = vb / (1.0 - beta2 ** t)

            # decoupled weight decay
            W *= (1.0 - lr * weight_decay)

            W -= lr * mW_hat / (np.sqrt(vW_hat) + eps)
            b -= lr * mb_hat / (np.sqrt(vb_hat) + eps)

        train_prob = softmax(X_train @ W + b)
        train_loss = cross_entropy(y_train, train_prob)
        train_acc = accuracy(y_train, train_prob)

        if X_valid is not None:
            valid_prob = softmax(X_valid @ W + b)
            valid_loss = cross_entropy(y_valid, valid_prob)
            valid_acc = accuracy(y_valid, valid_prob)

            if valid_acc > best_val_acc:
                best_val_acc = valid_acc
                best_epoch = epoch
                best_W = W.copy()
                best_b = b.copy()
                wait = 0
            else:
                wait += 1

            if epoch == 1 or epoch % 5 == 0:
                print(
                    f"Epoch {epoch:3d} | "
                    f"train_loss={train_loss:.4f} train_acc={train_acc:.4f} | "
                    f"valid_loss={valid_loss:.4f} valid_acc={valid_acc:.4f}"
                )

            if wait >= patience:
                print(f"Early stopping at epoch {epoch}")
                break
        else:
            if epoch == 1 or epoch % 5 == 0 or epoch == epochs:
                print(
                    f"Epoch {epoch:3d} | "
                    f"train_loss={train_loss:.4f} train_acc={train_acc:.4f}"
                )

    if X_valid is not None:
        return {
            "W": best_W,
            "b": best_b,
            "best_epoch": best_epoch,
            "best_val_acc": best_val_acc,
        }
    else:
        return {
            "W": W,
            "b": b,
            "best_epoch": epochs,
            "best_val_acc": None,
        }

# =========================
# 1) feature extraction
# =========================
X_all = make_features(x_train, batch_size=2000)
X_test_feat = make_features(x_test, batch_size=2000)

# =========================
# 2) validation split
# =========================
y_labels = np.argmax(y_train, axis=1)
idx = np.arange(len(X_all))

idx_tr, idx_va = train_test_split(
    idx,
    test_size=0.1,
    random_state=42,
    stratify=y_labels
)

X_tr_raw = X_all[idx_tr]
X_va_raw = X_all[idx_va]
y_tr = y_train[idx_tr]
y_va = y_train[idx_va]

mean, std = standardize_fit(X_tr_raw)
X_tr = standardize_apply(X_tr_raw, mean, std)
X_va = standardize_apply(X_va_raw, mean, std)

# =========================
# 3) stage 1: best epoch search
# =========================
stage1 = train_softmax(
    X_tr, y_tr,
    X_valid=X_va, y_valid=y_va,
    epochs=100,
    batch_size=256,
    lr_max=2e-3,
    lr_min=2e-4,
    weight_decay=1e-4,
    label_smoothing=0.02,
    patience=15,
    seed=42
)

print("best_epoch =", stage1["best_epoch"])
print("best_val_acc =", stage1["best_val_acc"])

# =========================
# 4) stage 2: retrain on full training set
# =========================
mean_full, std_full = standardize_fit(X_all)
X_full = standardize_apply(X_all, mean_full, std_full)
X_test_std = standardize_apply(X_test_feat, mean_full, std_full)

final_model = train_softmax(
    X_full, y_train,
    X_valid=None, y_valid=None,
    epochs=stage1["best_epoch"],
    batch_size=256,
    lr_max=2e-3,
    lr_min=2e-4,
    weight_decay=1e-4,
    label_smoothing=0.02,
    patience=15,
    seed=42
)

# =========================
# 5) prediction / submission
# =========================
test_prob = softmax(X_test_std @ final_model["W"] + final_model["b"])
y_pred = np.argmax(test_prob, axis=1)

submission = pd.Series(y_pred, name="label")
submission.to_csv(
    work_dir + "/Lecture02/submission_pred_02.csv",
    header=True,
    index_label="id"
)
